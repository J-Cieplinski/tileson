90 bool tson::Tile::parse(const nlohmann::json &json)
91 {
92  bool allFound = true;
93  #ifndef DISABLE_CPP17_FILESYSTEM
94  if(json.count("image") > 0) m_image = fs::path(json["image"].get<std::string>()); //Optional
95  #else
96  if(json.count("image") > 0) m_image = json["image"].get<std::string>(); //Optional
97  #endif
98  if(json.count("id") > 0) m_id = json["id"].get<int>() + 1; else allFound = false;
99  if(json.count("type") > 0) m_type = json["type"].get<std::string>(); //Optional
100  if(json.count("objectgroup") > 0) m_objectgroup = tson::Layer(json["objectgroup"]); //Optional
101 
102  if(json.count("imagewidth") > 0 && json.count("imageheight") > 0)
103  m_imageSize = {json["imagewidth"].get<int>(), json["imageheight"].get<int>()}; //Optional
104 
105  //More advanced data
106  if(json.count("animation") > 0 && json["animation"].is_array())
107  std::for_each(json["animation"].begin(), json["animation"].end(), [&](const nlohmann::json &item) { m_animation.emplace_back(item); });
108  if(json.count("terrain") > 0 && json["terrain"].is_array())
109  std::for_each(json["terrain"].begin(), json["terrain"].end(), [&](const nlohmann::json &item) { m_terrain.emplace_back(item.get<int>()); });
110 
111  if(json.count("properties") > 0 && json["properties"].is_array())
112  std::for_each(json["properties"].begin(), json["properties"].end(), [&](const nlohmann::json &item) { m_properties.add(item); });
113 
114  return allFound;
115 }
