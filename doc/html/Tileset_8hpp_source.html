119 bool tson::Tileset::parse(const nlohmann::json &json, tson::Map *map)
120 {
121  m_map = map;
122  bool allFound = true;
123 
124  if(json.count("columns") > 0) m_columns = json["columns"].get<int>(); else allFound = false;
125  if(json.count("firstgid") > 0) m_firstgid = json["firstgid"].get<int>(); else allFound = false;
126  #ifndef DISABLE_CPP17_FILESYSTEM
127  if(json.count("image") > 0) m_image = fs::path(json["image"].get<std::string>()); else allFound = false;
128  #else
129  if(json.count("image") > 0) m_image = json["image"].get<std::string>(); else allFound = false;
130  #endif
131  if(json.count("margin") > 0) m_margin = json["margin"].get<int>(); else allFound = false;
132  if(json.count("name") > 0) m_name = json["name"].get<std::string>(); else allFound = false;
133  if(json.count("spacing") > 0) m_spacing = json["spacing"].get<int>(); else allFound = false;
134  if(json.count("tilecount") > 0) m_tileCount = json["tilecount"].get<int>(); else allFound = false;
135  if(json.count("transparentcolor") > 0) m_transparentColor = tson::Colori(json["transparentcolor"].get<std::string>()); //Optional
136  if(json.count("type") > 0) m_type = json["type"].get<std::string>();
137  if(json.count("grid") > 0) m_grid = tson::Grid(json["grid"]);
138 
139  if(json.count("imagewidth") > 0 && json.count("imageheight") > 0)
140  m_imageSize = {json["imagewidth"].get<int>(), json["imageheight"].get<int>()}; else allFound = false;
141  if(json.count("tilewidth") > 0 && json.count("tileheight") > 0)
142  m_tileSize = {json["tilewidth"].get<int>(), json["tileheight"].get<int>()}; else allFound = false;
143  if(json.count("tileoffset") > 0)
144  m_tileOffset = {json["tileoffset"]["x"].get<int>(), json["tileoffset"]["y"].get<int>()};
145 
146  //More advanced data
147  if(json.count("wangsets") > 0 && json["wangsets"].is_array())
148  std::for_each(json["wangsets"].begin(), json["wangsets"].end(), [&](const nlohmann::json &item) { m_wangsets.emplace_back(item); });
149  if(json.count("tiles") > 0 && json["tiles"].is_array())
150  std::for_each(json["tiles"].begin(), json["tiles"].end(), [&](const nlohmann::json &item) { m_tiles.emplace_back(item, this, m_map); });
151  if(json.count("terrains") > 0 && json["terrains"].is_array())
152  std::for_each(json["terrains"].begin(), json["terrains"].end(), [&](const nlohmann::json &item) { m_terrains.emplace_back(item); });
153 
154  if(json.count("properties") > 0 && json["properties"].is_array())
155  std::for_each(json["properties"].begin(), json["properties"].end(), [&](const nlohmann::json &item) { m_properties.add(item); });
156 
157  if(json.count("objectalignment") > 0)
158  {
159  std::string alignment = json["objectalignment"].get<std::string>();
160  m_objectAlignment = StringToAlignment(alignment);
161  }
162 
163  generateMissingTiles();
164 
165  return allFound;
166 }

376 void tson::Tileset::generateMissingTiles()
377 {
378  std::vector<uint32_t> tileIds;
379  for(auto &tile : m_tiles)
380  tileIds.push_back(tile.getId());
381 
382  for(uint32_t i = m_firstgid; i < m_firstgid + m_tileCount; ++i)
383  {
384  if(std::count(tileIds.begin(), tileIds.end(), i) == 0)
385  {
386  m_tiles.emplace_back(Tile(i, this, m_map));
387  }
388  }
389 }
