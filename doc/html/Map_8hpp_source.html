134 bool tson::Map::parse(const nlohmann::json &json, tson::DecompressorContainer *decompressors)
135 {
136  m_decompressors = decompressors;
137 
138  bool allFound = true;
139  if(json.count("compressionlevel") > 0) m_compressionLevel = json["compressionlevel"].get<int>(); //Tiled 1.3 - Optional
140  if(json.count("backgroundcolor") > 0) m_backgroundColor = Colori(json["backgroundcolor"].get<std::string>()); //Optional
141  if(json.count("width") > 0 && json.count("height") > 0 )
142  m_size = {json["width"].get<int>(), json["height"].get<int>()}; else allFound = false;
143  if(json.count("hexsidelength") > 0) m_hexsideLength = json["hexsidelength"].get<int>(); //Optional
144  if(json.count("infinite") > 0) m_isInfinite = json["infinite"].get<bool>(); //Optional
145  if(json.count("nextlayerid") > 0) m_nextLayerId = json["nextlayerid"].get<int>(); //Optional
146  if(json.count("nextobjectid") > 0) m_nextObjectId = json["nextobjectid"].get<int>(); else allFound = false;
147  if(json.count("orientation") > 0) m_orientation = json["orientation"].get<std::string>(); else allFound = false;
148  if(json.count("renderorder") > 0) m_renderOrder = json["renderorder"].get<std::string>(); //Optional
149  if(json.count("staggeraxis") > 0) m_staggerAxis = json["staggeraxis"].get<std::string>(); //Optional
150  if(json.count("staggerindex") > 0) m_staggerIndex = json["staggerindex"].get<std::string>(); //Optional
151  if(json.count("tiledversion") > 0) m_tiledVersion = json["tiledversion"].get<std::string>(); else allFound = false;
152  if(json.count("tilewidth") > 0 && json.count("tileheight") > 0 )
153  m_tileSize = {json["tilewidth"].get<int>(), json["tileheight"].get<int>()}; else allFound = false;
154  if(json.count("type") > 0) m_type = json["type"].get<std::string>(); //Optional
155  if(json.count("version") > 0) m_version = json["version"].get<int>(); else allFound = false;
156 
157  //More advanced data
158  if(json.count("layers") > 0 && json["layers"].is_array())
159  std::for_each(json["layers"].begin(), json["layers"].end(), [&](const nlohmann::json &item) { m_layers.emplace_back(item, this); });
160  if(json.count("tilesets") > 0 && json["tilesets"].is_array())
161  std::for_each(json["tilesets"].begin(), json["tilesets"].end(), [&](const nlohmann::json &item) { m_tilesets.emplace_back(item, this); });
162  if(json.count("properties") > 0 && json["properties"].is_array())
163  std::for_each(json["properties"].begin(), json["properties"].end(), [&](const nlohmann::json &item) { m_properties.add(item); });
164 
165 
166  processData();
167 
168  return allFound;
169 }

174 void tson::Map::processData()
175 {
176  m_tileMap.clear();
177  for(auto &tileset : m_tilesets)
178  {
179  std::for_each(tileset.getTiles().begin(), tileset.getTiles().end(), [&](tson::Tile &tile) { m_tileMap[tile.getId()] = &tile; });
180  }
181  std::for_each(m_layers.begin(), m_layers.end(), [&](tson::Layer &layer)
182  {
183  layer.assignTileMap(&m_tileMap);
184  layer.createTileData(m_size, m_isInfinite);
185  const std::set<uint32_t> &flaggedTiles = layer.getUniqueFlaggedTiles();
186  for(uint32_t ftile : flaggedTiles)
187  {
188  tson::Tile tile {ftile, layer.getMap()};
189  if(m_tileMap.count(tile.getGid()))
190  {
191  tson::Tile *originalTile = m_tileMap[tile.getGid()];
192  tile.addTilesetAndPerformCalculations(originalTile->getTileset());
193  tile.setProperties(originalTile->getProperties());
194  m_flaggedTileMap[ftile] = tile;
195  m_tileMap[ftile] = &m_flaggedTileMap[ftile];
196  }
197  }
198  layer.resolveFlaggedTiles();
199  });
200 }
