135 bool tson::Map::parse(const nlohmann::json &json, tson::DecompressorContainer *decompressors)
136 {
137  m_decompressors = decompressors;
138 
139  bool allFound = true;
140  if(json.count("compressionlevel") > 0) m_compressionLevel = json["compressionlevel"].get<int>(); //Tiled 1.3 - Optional
141  if(json.count("backgroundcolor") > 0) m_backgroundColor = Colori(json["backgroundcolor"].get<std::string>()); //Optional
142  if(json.count("width") > 0 && json.count("height") > 0 )
143  m_size = {json["width"].get<int>(), json["height"].get<int>()}; else allFound = false;
144  if(json.count("hexsidelength") > 0) m_hexsideLength = json["hexsidelength"].get<int>(); //Optional
145  if(json.count("infinite") > 0) m_isInfinite = json["infinite"].get<bool>(); //Optional
146  if(json.count("nextlayerid") > 0) m_nextLayerId = json["nextlayerid"].get<int>(); //Optional
147  if(json.count("nextobjectid") > 0) m_nextObjectId = json["nextobjectid"].get<int>(); else allFound = false;
148  if(json.count("orientation") > 0) m_orientation = json["orientation"].get<std::string>(); else allFound = false;
149  if(json.count("renderorder") > 0) m_renderOrder = json["renderorder"].get<std::string>(); //Optional
150  if(json.count("staggeraxis") > 0) m_staggerAxis = json["staggeraxis"].get<std::string>(); //Optional
151  if(json.count("staggerindex") > 0) m_staggerIndex = json["staggerindex"].get<std::string>(); //Optional
152  if(json.count("tiledversion") > 0) m_tiledVersion = json["tiledversion"].get<std::string>(); else allFound = false;
153  if(json.count("tilewidth") > 0 && json.count("tileheight") > 0 )
154  m_tileSize = {json["tilewidth"].get<int>(), json["tileheight"].get<int>()}; else allFound = false;
155  if(json.count("type") > 0) m_type = json["type"].get<std::string>(); //Optional
156  if(json.count("version") > 0) m_version = json["version"].get<int>(); else allFound = false;
157 
158  //More advanced data
159  if(json.count("layers") > 0 && json["layers"].is_array())
160  std::for_each(json["layers"].begin(), json["layers"].end(), [&](const nlohmann::json &item) { m_layers.emplace_back(item, this); });
161 
162  if(json.count("properties") > 0 && json["properties"].is_array())
163  std::for_each(json["properties"].begin(), json["properties"].end(), [&](const nlohmann::json &item) { m_properties.add(item); });
164 
165  createTilesetData(json);
166  processData();
167 
168  return allFound;
169 }

174 void tson::Map::createTilesetData(const nlohmann::json &json)
175 {
176  if(json.count("tilesets") > 0 && json["tilesets"].is_array())
177  {
178  //First created tileset objects
179  std::for_each(json["tilesets"].begin(), json["tilesets"].end(), [&](const nlohmann::json &item)
180  {
181  m_tilesets.emplace_back();
182  });
183 
184  int i = 0;
185  //Then do the parsing
186  std::for_each(json["tilesets"].begin(), json["tilesets"].end(), [&](const nlohmann::json &item)
187  {
188  m_tilesets[i].parse(item, this);
189  ++i;
190  });
191  }

197 void tson::Map::processData()
198 {
199  m_tileMap.clear();
200  for(auto &tileset : m_tilesets)
201  {
202  std::for_each(tileset.getTiles().begin(), tileset.getTiles().end(), [&](tson::Tile &tile) { m_tileMap[tile.getId()] = &tile; });
203  }
204  std::for_each(m_layers.begin(), m_layers.end(), [&](tson::Layer &layer)
205  {
206  layer.assignTileMap(&m_tileMap);
207  layer.createTileData(m_size, m_isInfinite);
208  const std::set<uint32_t> &flaggedTiles = layer.getUniqueFlaggedTiles();
209  for(uint32_t ftile : flaggedTiles)
210  {
211  tson::Tile tile {ftile, layer.getMap()};
212  if(m_tileMap.count(tile.getGid()))
213  {
214  tson::Tile *originalTile = m_tileMap[tile.getGid()];
215  tile.addTilesetAndPerformCalculations(originalTile->getTileset());
216  tile.setProperties(originalTile->getProperties());
217  m_flaggedTileMap[ftile] = tile;
218  m_tileMap[ftile] = &m_flaggedTileMap[ftile];
219  }
220  }
221  layer.resolveFlaggedTiles();
222  });
223 }
