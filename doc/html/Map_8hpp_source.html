129 bool tson::Map::parse(const nlohmann::json &json)
130 {
131  bool allFound = true;
132  if(json.count("backgroundcolor") > 0) m_backgroundColor = Colori(json["backgroundcolor"].get<std::string>()); //Optional
133  if(json.count("width") > 0 && json.count("height") > 0 )
134  m_size = {json["width"].get<int>(), json["height"].get<int>()}; else allFound = false;
135  if(json.count("hexsidelength") > 0) m_hexsideLength = json["hexsidelength"].get<int>(); //Optional
136  if(json.count("infinite") > 0) m_isInfinite = json["infinite"].get<bool>(); //Optional
137  if(json.count("nextlayerid") > 0) m_nextLayerId = json["nextlayerid"].get<int>(); //Optional
138  if(json.count("nextobjectid") > 0) m_nextObjectId = json["nextobjectid"].get<int>(); else allFound = false;
139  if(json.count("orientation") > 0) m_orientation = json["orientation"].get<std::string>(); else allFound = false;
140  if(json.count("renderorder") > 0) m_renderOrder = json["renderorder"].get<std::string>(); //Optional
141  if(json.count("staggeraxis") > 0) m_staggerAxis = json["staggeraxis"].get<std::string>(); //Optional
142  if(json.count("staggerindex") > 0) m_staggerIndex = json["staggerindex"].get<std::string>(); //Optional
143  if(json.count("tiledversion") > 0) m_tiledVersion = json["tiledversion"].get<std::string>(); else allFound = false;
144  if(json.count("tilewidth") > 0 && json.count("tileheight") > 0 )
145  m_tileSize = {json["tilewidth"].get<int>(), json["tileheight"].get<int>()}; else allFound = false;
146  if(json.count("type") > 0) m_type = json["type"].get<std::string>(); //Optional
147  if(json.count("version") > 0) m_version = json["version"].get<int>(); else allFound = false;
148 
149  //More advanced data
150  if(json.count("layers") > 0 && json["layers"].is_array())
151  std::for_each(json["layers"].begin(), json["layers"].end(), [&](const nlohmann::json &item) { m_layers.emplace_back(item); });
152  if(json.count("tilesets") > 0 && json["tilesets"].is_array())
153  std::for_each(json["tilesets"].begin(), json["tilesets"].end(), [&](const nlohmann::json &item) { m_tilesets.emplace_back(item); });
154  if(json.count("properties") > 0 && json["properties"].is_array())
155  std::for_each(json["properties"].begin(), json["properties"].end(), [&](const nlohmann::json &item) { m_properties.add(item); });
156 
157 
158  processData();
159 
160  return allFound;
161 }

166 void tson::Map::processData()
167 {
168  m_tileMap.clear();
169  for(auto &tileset : m_tilesets)
170  {
171  std::for_each(tileset.getTiles().begin(), tileset.getTiles().end(), [&](tson::Tile &tile) { m_tileMap[tile.getId()] = &tile; });
172  }
173  std::for_each(m_layers.begin(), m_layers.end(), [&](tson::Layer &layer)
174  {
175  layer.assignTileMap(m_tileMap);
176  layer.createTileData(m_size, m_isInfinite);
177  });
178 }
