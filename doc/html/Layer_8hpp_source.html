139 bool tson::Layer::parse(const nlohmann::json &json)
140 {
141  bool allFound = true;
142  if(json.count("compression") > 0) m_compression = json["compression"].get<std::string>(); //Optional
143  if(json.count("draworder") > 0) m_drawOrder = json["draworder"].get<std::string>(); //Optional
144  if(json.count("encoding") > 0) m_encoding = json["encoding"].get<std::string>(); //Optional
145  if(json.count("id") > 0) m_id = json["id"].get<int>(); //Optional
146  if(json.count("image") > 0) m_image = json["image"].get<std::string>(); //Optional
147  if(json.count("name") > 0) m_name = json["name"].get<std::string>(); else allFound = false;
148  if(json.count("offsetx") > 0 && json.count("offsety") > 0)
149  m_offset = {json["offsetx"].get<float>(), json["offsety"].get<float>()}; //Optional
150  if(json.count("opacity") > 0) m_opacity = json["opacity"].get<float>(); else allFound = false;
151  if(json.count("width") > 0 && json.count("height") > 0)
152  m_size = {json["width"].get<int>(), json["height"].get<int>()}; else allFound = false;
153  if(json.count("transparentcolor") > 0) m_transparentcolor = tson::Colori(json["transparentcolor"].get<std::string>()); //Optional
154  if(json.count("type") > 0) m_typeStr = json["type"].get<std::string>(); else allFound = false;
155  if(json.count("visible") > 0) m_visible = json["visible"].get<bool>(); else allFound = false;
156  if(json.count("x") > 0) m_x = json["x"].get<int>(); else allFound = false;
157  if(json.count("y") > 0) m_y = json["y"].get<int>(); else allFound = false;
158 
159  //Handle DATA (Optional)
160  if(json.count("data") > 0)
161  {
162  if(json["data"].is_array())
163  {
164  std::for_each(json["data"].begin(), json["data"].end(), [&](const nlohmann::json &item) { m_data.push_back(item.get<int>()); });
165  }
166  else
167  m_base64Data = json["data"].get<std::string>();
168  }
169 
170  //More advanced data
171  if(json.count("chunks") > 0 && json["chunks"].is_array())
172  std::for_each(json["chunks"].begin(), json["chunks"].end(), [&](const nlohmann::json &item) { m_chunks.emplace_back(item); });
173  if(json.count("layers") > 0 && json["layers"].is_array())
174  std::for_each(json["layers"].begin(), json["layers"].end(), [&](const nlohmann::json &item) { m_layers.emplace_back(item); });
175  if(json.count("objects") > 0 && json["objects"].is_array())
176  std::for_each(json["objects"].begin(), json["objects"].end(), [&](const nlohmann::json &item) { m_objects.emplace_back(item); });
177  if(json.count("properties") > 0 && json["properties"].is_array())
178  std::for_each(json["properties"].begin(), json["properties"].end(), [&](const nlohmann::json &item) { m_properties.add(item); });
179 
180  setTypeByString();
181 
182  return allFound;
183 }

473 void tson::Layer::createTileData(const Vector2i &mapSize, bool isInfiniteMap)
474 {
475  size_t x = 0;
476  size_t y = 0;
477  if(!isInfiniteMap)
478  {
479  std::for_each(m_data.begin(), m_data.end(), [&](int tileId)
480  {
481  if (x == mapSize.x)
482  {
483  ++y;
484  x = 0;
485  }
486 
487  if (tileId > 0)
488  {
489  m_tileData[{x, y}] = m_tileMap[tileId];
490  }
491  x++;
492  });
493  }
494 }
