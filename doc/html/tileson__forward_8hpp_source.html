43  int columns = m_tileset->getColumns();

45  int lastId = (m_tileset->getFirstgid() + m_tileset->getTileCount()) - 1;
46 
47  if (getGid() >= firstId && getGid() <= lastId)
48  {
49  int baseTilePosition = ((int)getGid() - firstId);
50 
51  int tileModX = (baseTilePosition % columns);
52  int currentRow = (baseTilePosition / columns);
53  int offsetX = (tileModX != 0) ? ((tileModX) * m_map->getTileSize().x) : (0 * m_map->getTileSize().x);
54  int offsetY = (currentRow < rows-1) ? (currentRow * m_map->getTileSize().y) : ((rows-1) * m_map->getTileSize().y);
55 
56  m_drawingRect = { offsetX, offsetY, m_map->getTileSize().x, m_map->getTileSize().y };
57  }
58  else
