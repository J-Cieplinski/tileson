109 bool tson::Object::parse(const nlohmann::json &json)
110 {
111  bool allFound = true;
112 
113  if(json.count("ellipse") > 0) m_ellipse = json["ellipse"].get<bool>(); //Optional
114  if(json.count("gid") > 0) m_gid = json["gid"].get<int>(); //Optional
115  if(json.count("id") > 0) m_id = json["id"].get<int>(); else allFound = false;
116  if(json.count("name") > 0) m_name = json["name"].get<std::string>(); else allFound = false;
117  if(json.count("point") > 0) m_point = json["point"].get<bool>(); //Optional
118  if(json.count("rotation") > 0) m_rotation = json["rotation"].get<float>(); else allFound = false;
119  if(json.count("template") > 0) m_template = json["template"].get<std::string>(); //Optional
120  if(json.count("type") > 0) m_type = json["type"].get<std::string>(); else allFound = false;
121  if(json.count("visible") > 0) m_visible = json["visible"].get<bool>(); else allFound = false;
122 
123  if(json.count("width") > 0 && json.count("height") > 0)
124  m_size = {json["width"].get<int>(), json["height"].get<int>()}; else allFound = false;
125  if(json.count("x") > 0 && json.count("y") > 0)
126  m_position = {json["x"].get<int>(), json["y"].get<int>()}; else allFound = false;
127 
128  if(json.count("text") > 0)
129  m_text = {json["text"]["text"].get<std::string>(), json["text"]["wrap"].get<bool>()}; //Optional
130 
131  setObjectTypeByJson(json);
132 
133  if(m_objectType == ObjectType::Template)
134  allFound = true; //Just accept anything with this type
135 
136  //More advanced data
137  if(json.count("polygon") > 0 && json["polygon"].is_array())
138  std::for_each(json["polygon"].begin(), json["polygon"].end(),
139  [&](const nlohmann::json &item) { m_polygon.emplace_back(item["x"].get<int>(), item["y"].get<int>()); });
140 
141  if(json.count("polyline") > 0 && json["polyline"].is_array())
142  std::for_each(json["polyline"].begin(), json["polyline"].end(),
143  [&](const nlohmann::json &item) { m_polyline.emplace_back(item["x"].get<int>(), item["y"].get<int>()); });
144 
145  if(json.count("properties") > 0 && json["properties"].is_array())
146  std::for_each(json["properties"].begin(), json["properties"].end(), [&](const nlohmann::json &item) { m_properties.add(item); });
147 
148  return allFound;
149 }

156 void tson::Object::setObjectTypeByJson(const nlohmann::json &json)
157 {
158  m_objectType = ObjectType::Undefined;
159  if(m_ellipse)
160  m_objectType = ObjectType::Ellipse;
161  else if(m_point)
162  m_objectType = ObjectType::Point;
163  else if(json.count("polygon") > 0)
164  m_objectType = ObjectType::Polygon;
165  else if(json.count("polyline") > 0)
166  m_objectType = ObjectType::Polyline;
167  else if(json.count("text") > 0)
168  m_objectType = ObjectType::Text;
169  else if(json.count("gid") > 0)
170  m_objectType = ObjectType::Object;
171  else if(json.count("template") > 0)
172  m_objectType = ObjectType::Template;
173  else
174  m_objectType = ObjectType::Rectangle;
175 }
