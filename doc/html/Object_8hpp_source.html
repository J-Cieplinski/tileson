117 bool tson::Object::parse(const nlohmann::json &json)
118 {
119  bool allFound = true;
120 
121  if(json.count("ellipse") > 0) m_ellipse = json["ellipse"].get<bool>(); //Optional
122  if(json.count("gid") > 0)
123  {
124  uint32_t gid = json["gid"].get<uint32_t>(); //Optional
125  if (gid & FLIPPED_HORIZONTALLY_FLAG) m_flipFlags |= TileFlipFlags::Horizontally;
126  if (gid & FLIPPED_VERTICALLY_FLAG) m_flipFlags |= TileFlipFlags::Vertically;
127  if (gid & FLIPPED_DIAGONALLY_FLAG) m_flipFlags |= TileFlipFlags::Diagonally;
128 
129  // Clear flags
130  gid &= ~(FLIPPED_HORIZONTALLY_FLAG | FLIPPED_VERTICALLY_FLAG | FLIPPED_DIAGONALLY_FLAG);
131 
132  m_gid = gid;
133  }
134  if(json.count("id") > 0) m_id = json["id"].get<int>(); else allFound = false;
135  if(json.count("name") > 0) m_name = json["name"].get<std::string>(); else allFound = false;
136  if(json.count("point") > 0) m_point = json["point"].get<bool>(); //Optional
137  if(json.count("rotation") > 0) m_rotation = json["rotation"].get<float>(); else allFound = false;
138  if(json.count("template") > 0) m_template = json["template"].get<std::string>(); //Optional
139  if(json.count("type") > 0) m_type = json["type"].get<std::string>(); else allFound = false;
140  if(json.count("visible") > 0) m_visible = json["visible"].get<bool>(); else allFound = false;
141 
142  if(json.count("width") > 0 && json.count("height") > 0)
143  m_size = {json["width"].get<int>(), json["height"].get<int>()}; else allFound = false;
144  if(json.count("x") > 0 && json.count("y") > 0)
145  m_position = {json["x"].get<int>(), json["y"].get<int>()}; else allFound = false;
146 
147  if(json.count("text") > 0)
148  m_text = {json["text"]["text"].get<std::string>(), json["text"]["wrap"].get<bool>(), tson::Colori(json["text"]["color"].get<std::string>())}; //Optional
149 
150  setObjectTypeByJson(json);
151 
152  if(m_objectType == ObjectType::Template)
153  allFound = true; //Just accept anything with this type
154 
155  //More advanced data
156  if(json.count("polygon") > 0 && json["polygon"].is_array())
157  std::for_each(json["polygon"].begin(), json["polygon"].end(),
158  [&](const nlohmann::json &item) { m_polygon.emplace_back(item["x"].get<int>(), item["y"].get<int>()); });
159 
160  if(json.count("polyline") > 0 && json["polyline"].is_array())
161  std::for_each(json["polyline"].begin(), json["polyline"].end(),
162  [&](const nlohmann::json &item) { m_polyline.emplace_back(item["x"].get<int>(), item["y"].get<int>()); });
163 
164  if(json.count("properties") > 0 && json["properties"].is_array())
165  std::for_each(json["properties"].begin(), json["properties"].end(), [&](const nlohmann::json &item) { m_properties.add(item); });
166 
167  return allFound;
168 }

175 void tson::Object::setObjectTypeByJson(const nlohmann::json &json)
176 {
177  m_objectType = ObjectType::Undefined;
178  if(m_ellipse)
179  m_objectType = ObjectType::Ellipse;
180  else if(m_point)
181  m_objectType = ObjectType::Point;
182  else if(json.count("polygon") > 0)
183  m_objectType = ObjectType::Polygon;
184  else if(json.count("polyline") > 0)
185  m_objectType = ObjectType::Polyline;
186  else if(json.count("text") > 0)
187  m_objectType = ObjectType::Text;
188  else if(json.count("gid") > 0)
189  m_objectType = ObjectType::Object;
190  else if(json.count("template") > 0)
191  m_objectType = ObjectType::Template;
192  else
193  m_objectType = ObjectType::Rectangle;
194 }
