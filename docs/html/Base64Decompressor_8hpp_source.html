31  std::string Base64Decompressor::decompress(std::string_view s)
32  {
33 
34  size_t length_of_string = s.length();
35  if (!length_of_string) return std::string("");
36 
37  size_t in_len = length_of_string;
38  size_t pos = 0;
39 
40  //
41  // The approximate length (bytes) of the decoded string might be one ore
42  // two bytes smaller, depending on the amount of trailing equal signs
43  // in the encoded string. This approximation is needed to reserve
44  // enough space in the string to be returned.
45  //
46  size_t approx_length_of_decoded_string = length_of_string / 4 * 3;
47  std::string ret;
48  ret.reserve(approx_length_of_decoded_string);
49 
50  while (pos < in_len) {
51 
52  unsigned int pos_of_char_1 = pos_of_char(s[pos+1] );
53 
54  ret.push_back(static_cast<std::string::value_type>( ( (pos_of_char(s[pos+0]) ) << 2 ) + ( (pos_of_char_1 & 0x30 ) >> 4)));
55 
56  if (s[pos+2] != '=' && s[pos+2] != '.') { // accept URL-safe base 64 strings, too, so check for '.' also.
57 
58  unsigned int pos_of_char_2 = pos_of_char(s[pos+2] );
59  ret.push_back(static_cast<std::string::value_type>( (( pos_of_char_1 & 0x0f) << 4) + (( pos_of_char_2 & 0x3c) >> 2)));
60 
61  if (s[pos+3] != '=' && s[pos+3] != '.') {
62  ret.push_back(static_cast<std::string::value_type>( ( (pos_of_char_2 & 0x03 ) << 6 ) + pos_of_char(s[pos+3]) ));
63  }
64  }
65 
66  pos += 4;
67  }
68

70  }
