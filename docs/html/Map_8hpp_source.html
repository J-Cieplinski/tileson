149bool tson::Map::parse(IJson &json, tson::DecompressorContainer *decompressors, tson::Project *project)
150{
151 m_decompressors = decompressors;
152 m_project = project;
153
154 bool allFound = true;
155 if(json.count("compressionlevel") > 0)
156 m_compressionLevel = json["compressionlevel"].get<int>(); //Tiled 1.3 - Optional
157
158 if(json.count("backgroundcolor") > 0) m_backgroundColor = Colori(json["backgroundcolor"].get<std::string>()); //Optional
159 if(json.count("width") > 0 && json.count("height") > 0 )
160 m_size = {json["width"].get<int>(), json["height"].get<int>()}; else allFound = false;
161 if(json.count("hexsidelength") > 0) m_hexsideLength = json["hexsidelength"].get<int>(); //Optional
162 if(json.count("infinite") > 0) m_isInfinite = json["infinite"].get<bool>(); //Optional
163 if(json.count("nextlayerid") > 0) m_nextLayerId = json["nextlayerid"].get<int>(); //Optional
164 if(json.count("nextobjectid") > 0) m_nextObjectId = json["nextobjectid"].get<int>(); else allFound = false;
165 if(json.count("orientation") > 0) m_orientation = json["orientation"].get<std::string>(); else allFound = false;
166 if(json.count("renderorder") > 0) m_renderOrder = json["renderorder"].get<std::string>(); //Optional
167 if(json.count("staggeraxis") > 0) m_staggerAxis = json["staggeraxis"].get<std::string>(); //Optional
168 if(json.count("staggerindex") > 0) m_staggerIndex = json["staggerindex"].get<std::string>(); //Optional
169 if(json.count("tiledversion") > 0) m_tiledVersion = json["tiledversion"].get<std::string>(); else allFound = false;
170 if(json.count("tilewidth") > 0 && json.count("tileheight") > 0 )
171 m_tileSize = {json["tilewidth"].get<int>(), json["tileheight"].get<int>()}; else allFound = false;
172 if(json.count("type") > 0) m_type = json["type"].get<std::string>(); //Optional
173 if(json.count("class") > 0) m_classType = json["class"].get<std::string>(); //Optional
174
175 //Removed - Changed from a float to string in Tiled v1.6, and old spec said int.
176 //Reason for removal is that it seems to have no real use, as TiledVersion is stored in another variable.
177 //if(json.count("version") > 0) m_version = json["version"].get<int>(); else allFound = false;
178
179 //More advanced data
180 if(json.count("layers") > 0 && json["layers"].isArray())
181 {
182 auto &array = json.array("layers");
183 std::for_each(array.begin(), array.end(), [&](std::unique_ptr<IJson> &item)
184 {
185 m_layers.emplace_back(*item, this);
186 });
187 }
188
189 if(json.count("properties") > 0 && json["properties"].isArray())
190 {
191 auto &array = json.array("properties");
192 std::for_each(array.begin(), array.end(), [&](std::unique_ptr<IJson> &item)
193 {
194 m_properties.add(*item, m_project);
195 });
196 }
197
198 tson::Vector2f parallaxOrigin {0.f, 0.f};
199 if(json.count("parallaxoriginx") > 0)
200 parallaxOrigin.x = json["parallaxoriginx"].get<float>();
201 if(json.count("parallaxoriginy") > 0)
202 parallaxOrigin.y = json["parallaxoriginy"].get<float>();
203
204 m_parallaxOrigin = parallaxOrigin;
205
206 if(!createTilesetData(json))
207 allFound = false;
208
209 processData();
210
211 return allFound;
212}

217bool tson::Map::createTilesetData(IJson &json)
218{
219 bool ok = true;
220 if(json.count("tilesets") > 0 && json["tilesets"].isArray())
221 {
222 //First created tileset objects
223 auto &tilesets = json.array("tilesets");
224 std::for_each(tilesets.begin(), tilesets.end(), [&](std::unique_ptr<IJson> &item)
225 {
226 m_tilesets.emplace_back();
227 });
228
229 int i = 0;
230 //Then do the parsing
231 std::for_each(tilesets.begin(), tilesets.end(), [&](std::unique_ptr<IJson> &item)
232 {
233 item->directory(json.directory());
234 if(!m_tilesets[i].parse(*item, this))
235 ok = false;
236
237 ++i;
238 });
239 }
240 return ok;
241}

246void tson::Map::processData()
247{
248 m_tileMap.clear();
249 for(auto &tileset : m_tilesets)
250 {
251 std::set<std::uint32_t> usedIds;
252 for(auto& tile : tileset.getTiles())
253 {
254 if (usedIds.count(tile.getGid()) != 0)
255 {
256 continue;
257 }
258 usedIds.insert(tile.getGid());
259 m_tileMap[tile.getGid()] = &tile;
260 }
261 }
262 std::for_each(m_layers.begin(), m_layers.end(), [&](tson::Layer &layer)
263 {
264 layer.assignTileMap(&m_tileMap);
265 layer.createTileData(m_size, m_isInfinite);
266 const std::set<uint32_t> &flaggedTiles = layer.getUniqueFlaggedTiles();
267 for(uint32_t ftile : flaggedTiles)
268 {
269 tson::Tile tile {ftile, layer.getMap()};
270 if(m_tileMap.count(tile.getGid()))
271 {
272 tson::Tile *originalTile = m_tileMap[tile.getGid()];
273 tile.addTilesetAndPerformCalculations(originalTile->getTileset());
274 tile.setProperties(originalTile->getProperties());
275 m_flaggedTileMap[ftile] = tile;
276 m_tileMap[ftile] = &m_flaggedTileMap[ftile];
277 }
278 }
279 layer.resolveFlaggedTiles();
280 });
281}

466tson::Tileset *tson::Map::getTilesetByGid(uint32_t gid)
467{
468 auto result = std::find_if(m_tilesets.begin(), m_tilesets.end(), [&](const tson::Tileset &tileset)
469 {
470 auto const firstId = static_cast<uint32_t>(tileset.getFirstgid()); //First tile id of the tileset
471 auto const lastId = static_cast<uint32_t>((firstId + tileset.getTileCount()) - 1);
472
473 return (gid >= firstId && gid <= lastId);
474 });
475 if(result == m_tilesets.end())
476 return nullptr;
477
478 return &result.operator*();
479}
