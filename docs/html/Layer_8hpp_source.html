161 bool tson::Layer::parse(IJson &json, tson::Map *map)
162 {
163  m_map = map;
164 
165  bool allFound = true;
166  if(json.count("tintcolor") > 0) m_tintcolor = tson::Colori(json["tintcolor"].get<std::string>()); //Optional
167  if(json.count("compression") > 0) m_compression = json["compression"].get<std::string>(); //Optional
168  if(json.count("draworder") > 0) m_drawOrder = json["draworder"].get<std::string>(); //Optional
169  if(json.count("encoding") > 0) m_encoding = json["encoding"].get<std::string>(); //Optional
170  if(json.count("id") > 0) m_id = json["id"].get<int>(); //Optional
171  if(json.count("image") > 0) m_image = json["image"].get<std::string>(); //Optional
172  if(json.count("name") > 0) m_name = json["name"].get<std::string>(); else allFound = false;
173  if(json.count("offsetx") > 0 && json.count("offsety") > 0)
174  m_offset = {json["offsetx"].get<float>(), json["offsety"].get<float>()}; //Optional
175  if(json.count("opacity") > 0) m_opacity = json["opacity"].get<float>(); else allFound = false;
176  if(json.count("width") > 0 && json.count("height") > 0)
177  m_size = {json["width"].get<int>(), json["height"].get<int>()}; //else allFound = false; - Not mandatory for all layers!
178  if(json.count("transparentcolor") > 0) m_transparentcolor = tson::Colori(json["transparentcolor"].get<std::string>()); //Optional
179  if(json.count("type") > 0) m_typeStr = json["type"].get<std::string>(); else allFound = false;
180  if(json.count("visible") > 0) m_visible = json["visible"].get<bool>(); else allFound = false;
181  if(json.count("x") > 0) m_x = json["x"].get<int>(); else allFound = false;
182  if(json.count("y") > 0) m_y = json["y"].get<int>(); else allFound = false;
183 
184  //Handle DATA (Optional)
185  if(json.count("data") > 0)
186  {
187  if(json["data"].isArray())
188  {
189  auto &array = json.array("data");
190  std::for_each(array.begin(), array.end(), [&](std::unique_ptr<IJson> &item) { m_data.push_back(item->get<uint32_t>()); });
191  }
192  else
193  {
194  m_base64Data = json["data"].get<std::string>();
195  decompressData();
196  }
197  }
198 
199  //More advanced data
200  if(json.count("chunks") > 0 && json["chunks"].isArray())
201  {
202  auto &chunks = json.array("chunks");
203  std::for_each(chunks.begin(), chunks.end(), [&](std::unique_ptr<IJson> &item) { m_chunks.emplace_back(*item); });
204  }
205  if(json.count("layers") > 0 && json["layers"].isArray())
206  {
207  auto &layers = json.array("layers");
208  std::for_each(layers.begin(), layers.end(), [&](std::unique_ptr<IJson> &item) { m_layers.emplace_back(*item, m_map); });
209  }
210  if(json.count("objects") > 0 && json["objects"].isArray())
211  {
212  auto &objects = json.array("objects");
213  std::for_each(objects.begin(), objects.end(), [&](std::unique_ptr<IJson> &item) { m_objects.emplace_back(*item); });
214  }
215  if(json.count("properties") > 0 && json["properties"].isArray())
216  {
217  auto &properties = json.array("properties");
218  std::for_each(properties.begin(), properties.end(), [&](std::unique_ptr<IJson> &item) { m_properties.add(*item); });
219  }
220 
221  setTypeByString();
222 
223  return allFound;
224 }

562 void tson::Layer::createTileData(const Vector2i &mapSize, bool isInfiniteMap)
563 {
564  size_t x = 0;
565  size_t y = 0;
566  if(!isInfiniteMap)
567  {
568  std::for_each(m_data.begin(), m_data.end(), [&](uint32_t tileId)
569  {
570  if (x == mapSize.x)
571  {
572  ++y;
573  x = 0;
574  }
575 
576  if (tileId > 0 && m_tileMap->count(tileId) > 0)
577  {
578  m_tileData[{x, y}] = m_tileMap->at(tileId);
579  m_tileObjects[{x, y}] = {{x, y}, m_tileData[{x, y}]};
580  }
581  else if(tileId > 0 && m_tileMap->count(tileId) == 0) //Tile with flip flags!
582  {
583  queueFlaggedTile(x, y, tileId);
584  }
585  x++;
586  });
587 
588  }
589 }
