161 bool tson::Layer::parse(const nlohmann::json &json, tson::Map *map)
162 {
163  m_map = map;
164 
165  bool allFound = true;
166  if(json.count("tintcolor") > 0) m_tintcolor = tson::Colori(json["tintcolor"].get<std::string>()); //Optional
167  if(json.count("compression") > 0) m_compression = json["compression"].get<std::string>(); //Optional
168  if(json.count("draworder") > 0) m_drawOrder = json["draworder"].get<std::string>(); //Optional
169  if(json.count("encoding") > 0) m_encoding = json["encoding"].get<std::string>(); //Optional
170  if(json.count("id") > 0) m_id = json["id"].get<int>(); //Optional
171  if(json.count("image") > 0) m_image = json["image"].get<std::string>(); //Optional
172  if(json.count("name") > 0) m_name = json["name"].get<std::string>(); else allFound = false;
173  if(json.count("offsetx") > 0 && json.count("offsety") > 0)
174  m_offset = {json["offsetx"].get<float>(), json["offsety"].get<float>()}; //Optional
175  if(json.count("opacity") > 0) m_opacity = json["opacity"].get<float>(); else allFound = false;
176  if(json.count("width") > 0 && json.count("height") > 0)
177  m_size = {json["width"].get<int>(), json["height"].get<int>()}; else allFound = false;
178  if(json.count("transparentcolor") > 0) m_transparentcolor = tson::Colori(json["transparentcolor"].get<std::string>()); //Optional
179  if(json.count("type") > 0) m_typeStr = json["type"].get<std::string>(); else allFound = false;
180  if(json.count("visible") > 0) m_visible = json["visible"].get<bool>(); else allFound = false;
181  if(json.count("x") > 0) m_x = json["x"].get<int>(); else allFound = false;
182  if(json.count("y") > 0) m_y = json["y"].get<int>(); else allFound = false;
183 
184  //Handle DATA (Optional)
185  if(json.count("data") > 0)
186  {
187  if(json["data"].is_array())
188  {
189  std::for_each(json["data"].begin(), json["data"].end(), [&](const nlohmann::json &item) { m_data.push_back(item.get<uint32_t>()); });
190  }
191  else
192  {
193  m_base64Data = json["data"].get<std::string>();
194  decompressData();
195  }
196  }
197 
198  //More advanced data
199  if(json.count("chunks") > 0 && json["chunks"].is_array())
200  std::for_each(json["chunks"].begin(), json["chunks"].end(), [&](const nlohmann::json &item) { m_chunks.emplace_back(item); });
201  if(json.count("layers") > 0 && json["layers"].is_array())
202  std::for_each(json["layers"].begin(), json["layers"].end(), [&](const nlohmann::json &item) { m_layers.emplace_back(item, m_map); });
203  if(json.count("objects") > 0 && json["objects"].is_array())
204  std::for_each(json["objects"].begin(), json["objects"].end(), [&](const nlohmann::json &item) { m_objects.emplace_back(item); });
205  if(json.count("properties") > 0 && json["properties"].is_array())
206  std::for_each(json["properties"].begin(), json["properties"].end(), [&](const nlohmann::json &item) { m_properties.add(item); });
207 
208  setTypeByString();
209 
210  return allFound;
211 }

549 void tson::Layer::createTileData(const Vector2i &mapSize, bool isInfiniteMap)
550 {
551  size_t x = 0;
552  size_t y = 0;
553  if(!isInfiniteMap)
554  {
555  std::for_each(m_data.begin(), m_data.end(), [&](uint32_t tileId)
556  {
557  if (x == mapSize.x)
558  {
559  ++y;
560  x = 0;
561  }
562 
563  if (tileId > 0 && m_tileMap->count(tileId) > 0)
564  {
565  m_tileData[{x, y}] = m_tileMap->at(tileId);
566  m_tileObjects[{x, y}] = {{x, y}, m_tileData[{x, y}]};
567  }
568  else if(tileId > 0 && m_tileMap->count(tileId) == 0) //Tile with flip flags!
569  {
570  queueFlaggedTile(x, y, tileId);
571  }
572  x++;
573  });
574 
575  }
576 }
