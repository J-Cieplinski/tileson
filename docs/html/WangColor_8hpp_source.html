53bool tson::WangColor::parse(IJson &json, tson::Map *map)
54{
55 m_map = map;
56 bool allFound = true;
57
58 if(json.count("color") > 0) m_color = tson::Colori(json["color"].get<std::string>()); else allFound = false;
59 if(json.count("name") > 0) m_name = json["name"].get<std::string>(); else allFound = false;
60 if(json.count("probability") > 0) m_probability = json["probability"].get<float>(); else allFound = false;
61 if(json.count("tile") > 0) m_tile = json["tile"].get<int>(); else allFound = false;
62 if(json.count("class") > 0) m_classType = json["class"].get<std::string>(); //Optional
63
64 if(json.count("properties") > 0 && json["properties"].isArray())
65 {
66 auto &properties = json.array("properties");
67 std::for_each(properties.begin(), properties.end(), [&](std::unique_ptr<IJson> &item) { m_properties.add(*item); });
68 }
69
70
71 return allFound;
72}
