147bool tson::Tile::parse(IJson &json, tson::Tileset *tileset, tson::Map *map)
148{
149 m_tileset = tileset;
150 m_map = map;
151
152 if(json.count("image") > 0) m_image = fs::path(json["image"].get<std::string>()); //Optional
153
154 bool allFound = parseId(json);
155
156 if(json.count("type") > 0) m_type = json["type"].get<std::string>(); //Optional
157 else if(json.count("class") > 0) m_type = json["class"].get<std::string>(); //Tiled v1.9 renamed 'type' to 'class'
158
159 if(json.count("objectgroup") > 0) m_objectgroup = tson::Layer(json["objectgroup"], m_map); //Optional
160
161 if(json.count("imagewidth") > 0 && json.count("imageheight") > 0)
162 m_imageSize = {json["imagewidth"].get<int>(), json["imageheight"].get<int>()}; //Optional
163
164 m_subRect = {0,0, m_imageSize.x, m_imageSize.y};
165 if(json.count("x") > 0) m_subRect.x = json["x"].get<int>(); //Optional
166 if(json.count("y") > 0) m_subRect.y = json["y"].get<int>(); //Optional
167 if(json.count("width") > 0) m_subRect.width = json["width"].get<int>(); //Optional
168 if(json.count("height") > 0) m_subRect.height = json["height"].get<int>(); //Optional
169
170 //More advanced data
171 if(json.count("animation") > 0 && json["animation"].isArray())
172 {
173 auto &animation = json.array("animation");
174 std::vector<tson::Frame> frames;
175 std::for_each(animation.begin(), animation.end(), [&](std::unique_ptr<IJson> &item) { frames.emplace_back(*item); });
176 if(frames.size() > 0)
177 {
178 m_animation.setFrames(frames);
179 }
180 }
181 if(json.count("terrain") > 0 && json["terrain"].isArray())
182 {
183 auto &terrain = json.array("terrain");
184 std::for_each(terrain.begin(), terrain.end(), [&](std::unique_ptr<IJson> &item) { m_terrain.emplace_back(item->get<int>()); });
185 }
186
187 if(json.count("properties") > 0 && json["properties"].isArray())
188 {
189 auto &properties = json.array("properties");
190 std::for_each(properties.begin(), properties.end(), [&](std::unique_ptr<IJson> &item) { m_properties.add(*item); });
191 }
192
193 performDataCalculations();
194
195 return allFound;
196}

334void tson::Tile::manageFlipFlagsByIdThenRemoveFlags(uint32_t &id)
335{
336 if (id & FLIPPED_HORIZONTALLY_FLAG) m_flipFlags |= TileFlipFlags::Horizontally;
337 if (id & FLIPPED_VERTICALLY_FLAG) m_flipFlags |= TileFlipFlags::Vertically;
338 if (id & FLIPPED_DIAGONALLY_FLAG) m_flipFlags |= TileFlipFlags::Diagonally;
339
340 id &= ~(FLIPPED_HORIZONTALLY_FLAG | FLIPPED_VERTICALLY_FLAG | FLIPPED_DIAGONALLY_FLAG);
341}
