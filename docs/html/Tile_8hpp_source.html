142 bool tson::Tile::parse(IJson &json, tson::Tileset *tileset, tson::Map *map)
143 {
144  m_tileset = tileset;
145  m_map = map;
146 
147  if(json.count("image") > 0) m_image = fs::path(json["image"].get<std::string>()); //Optional
148 
149  bool allFound = parseId(json);
150 
151  if(json.count("type") > 0) m_type = json["type"].get<std::string>(); //Optional
152  if(json.count("objectgroup") > 0) m_objectgroup = tson::Layer(json["objectgroup"], m_map); //Optional
153 
154  if(json.count("imagewidth") > 0 && json.count("imageheight") > 0)
155  m_imageSize = {json["imagewidth"].get<int>(), json["imageheight"].get<int>()}; //Optional
156 
157  //More advanced data
158  if(json.count("animation") > 0 && json["animation"].isArray())
159  {
160  auto &animation = json.array("animation");
161  std::for_each(animation.begin(), animation.end(), [&](std::unique_ptr<IJson> &item) { m_animation.emplace_back(*item); });
162  }
163  if(json.count("terrain") > 0 && json["terrain"].isArray())
164  {
165  auto &terrain = json.array("terrain");
166  std::for_each(terrain.begin(), terrain.end(), [&](std::unique_ptr<IJson> &item) { m_terrain.emplace_back(item->get<int>()); });
167  }
168 
169  if(json.count("properties") > 0 && json["properties"].isArray())
170  {
171  auto &properties = json.array("properties");
172  std::for_each(properties.begin(), properties.end(), [&](std::unique_ptr<IJson> &item) { m_properties.add(*item); });
173  }
174 
175  performDataCalculations();
176 
177  return allFound;
178 }

305 void tson::Tile::manageFlipFlagsByIdThenRemoveFlags(uint32_t &id)
306 {
307  if (id & FLIPPED_HORIZONTALLY_FLAG) m_flipFlags |= TileFlipFlags::Horizontally;
308  if (id & FLIPPED_VERTICALLY_FLAG) m_flipFlags |= TileFlipFlags::Vertically;
309  if (id & FLIPPED_DIAGONALLY_FLAG) m_flipFlags |= TileFlipFlags::Diagonally;
310 
311  id &= ~(FLIPPED_HORIZONTALLY_FLAG | FLIPPED_VERTICALLY_FLAG | FLIPPED_DIAGONALLY_FLAG);
312 }
