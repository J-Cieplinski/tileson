123bool tson::Object::parse(IJson &json, tson::Map *map)
124{
125 m_map = map;
126 bool allFound = true;
127
128 if(json.count("ellipse") > 0) m_ellipse = json["ellipse"].get<bool>(); //Optional
129 if(json.count("gid") > 0)
130 {
131 uint32_t gid = json["gid"].get<uint32_t>(); //Optional
132 if (gid & FLIPPED_HORIZONTALLY_FLAG) m_flipFlags |= TileFlipFlags::Horizontally;
133 if (gid & FLIPPED_VERTICALLY_FLAG) m_flipFlags |= TileFlipFlags::Vertically;
134 if (gid & FLIPPED_DIAGONALLY_FLAG) m_flipFlags |= TileFlipFlags::Diagonally;
135
136 // Clear flags
137 gid &= ~(FLIPPED_HORIZONTALLY_FLAG | FLIPPED_VERTICALLY_FLAG | FLIPPED_DIAGONALLY_FLAG);
138
139 m_gid = gid;
140 }
141 if(json.count("id") > 0) m_id = json["id"].get<int>(); else allFound = false;
142 if(json.count("name") > 0) m_name = json["name"].get<std::string>(); else allFound = false;
143 if(json.count("point") > 0) m_point = json["point"].get<bool>(); //Optional
144 if(json.count("rotation") > 0) m_rotation = json["rotation"].get<float>(); else allFound = false;
145 if(json.count("template") > 0) m_template = json["template"].get<std::string>(); //Optional
146
147 if(json.count("type") > 0) m_type = json["type"].get<std::string>();
148 else if(json.count("class") > 0) m_type = json["class"].get<std::string>(); //Tiled v1.9 renamed 'type' to 'class'
149 else allFound = false;
150
151 if(json.count("visible") > 0) m_visible = json["visible"].get<bool>(); else allFound = false;
152
153 if(json.count("width") > 0 && json.count("height") > 0)
154 m_size = {json["width"].get<int>(), json["height"].get<int>()}; else allFound = false;
155 if(json.count("x") > 0 && json.count("y") > 0)
156 m_position = {json["x"].get<int>(), json["y"].get<int>()}; else allFound = false;
157
158 if(json.count("text") > 0)
159 {
160 bool hasColor = json["text"].count("color") > 0;
161 tson::Color c = (hasColor) ? tson::Colori(json["text"]["color"].get<std::string>()) : tson::Colori();
162 m_text = {json["text"]["text"].get<std::string>(), json["text"]["wrap"].get<bool>(), c}; //Optional
163 }
164
165 setObjectTypeByJson(json);
166
167 if(m_objectType == ObjectType::Template)
168 allFound = true; //Just accept anything with this type
169
170 //More advanced data
171 if(json.count("polygon") > 0 && json["polygon"].isArray())
172 {
173 auto &polygon = json.array("polygon");
174 std::for_each(polygon.begin(), polygon.end(),[&](std::unique_ptr<IJson> &item)
175 {
176 IJson &j = *item;
177 m_polygon.emplace_back(j["x"].get<int>(), j["y"].get<int>());
178 });
179
180 }
181
182 if(json.count("polyline") > 0 && json["polyline"].isArray())
183 {
184 auto &polyline = json.array("polyline");
185 std::for_each(polyline.begin(), polyline.end(),[&](std::unique_ptr<IJson> &item)
186 {
187 IJson &j = *item;
188 m_polyline.emplace_back(j["x"].get<int>(), j["y"].get<int>());
189 });
190 }
191
192 if(json.count("properties") > 0 && json["properties"].isArray())
193 {
194 auto &properties = json.array("properties");
195 std::for_each(properties.begin(), properties.end(), [&](std::unique_ptr<IJson> &item)
196 {
197 m_properties.add(*item);
198 });
199 }
200
201 return allFound;
202}

209void tson::Object::setObjectTypeByJson(IJson &json)
210{
211 m_objectType = ObjectType::Undefined;
212 if(m_ellipse)
213 m_objectType = ObjectType::Ellipse;
214 else if(m_point)
215 m_objectType = ObjectType::Point;
216 else if(json.count("polygon") > 0)
217 m_objectType = ObjectType::Polygon;
218 else if(json.count("polyline") > 0)
219 m_objectType = ObjectType::Polyline;
220 else if(json.count("text") > 0)
221 m_objectType = ObjectType::Text;
222 else if(json.count("gid") > 0)
223 m_objectType = ObjectType::Object;
224 else if(json.count("template") > 0)
225 m_objectType = ObjectType::Template;
226 else
227 m_objectType = ObjectType::Rectangle;
228}
