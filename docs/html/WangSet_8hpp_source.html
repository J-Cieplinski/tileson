63 bool tson::WangSet::parse(IJson &json)
64 {
65  bool allFound = true;
66 
67  if(json.count("tile") > 0) m_tile = json["tile"].get<int>(); else allFound = false;
68  if(json.count("name") > 0) m_name = json["name"].get<std::string>(); else allFound = false;
69 
70  //More advanced data
71  if(json.count("wangtiles") > 0 && json["wangtiles"].isArray())
72  {
73  auto &wangtiles = json.array("wangtiles");
74  std::for_each(wangtiles.begin(), wangtiles.end(), [&](std::unique_ptr<IJson> &item) { m_wangTiles.emplace_back(*item); });
75  }
76  if(json.count("cornercolors") > 0 && json["cornercolors"].isArray())
77  {
78  auto &cornercolors = json.array("cornercolors");
79  std::for_each(cornercolors.begin(), cornercolors.end(), [&](std::unique_ptr<IJson> &item) { m_cornerColors.emplace_back(*item); });
80  }
81  if(json.count("edgecolors") > 0 && json["edgecolors"].isArray())
82  {
83  auto &edgecolors = json.array("edgecolors");
84  std::for_each(edgecolors.begin(), edgecolors.end(), [&](std::unique_ptr<IJson> &item) { m_edgeColors.emplace_back(*item); });
85  }
86  if(json.count("properties") > 0 && json["properties"].isArray())
87  {
88  auto &properties = json.array("properties");
89  std::for_each(properties.begin(), properties.end(), [&](std::unique_ptr<IJson> &item) { m_properties.add(*item); });
90  }
91 
92  return allFound;
93 }
