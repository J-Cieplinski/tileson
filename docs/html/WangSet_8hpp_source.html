63 bool tson::WangSet::parse(const nlohmann::json &json)
64 {
65  bool allFound = true;
66 
67  if(json.count("tile") > 0) m_tile = json["tile"].get<int>(); else allFound = false;
68  if(json.count("name") > 0) m_name = json["name"].get<std::string>(); else allFound = false;
69 
70  //More advanced data
71  if(json.count("wangtiles") > 0 && json["wangtiles"].is_array())
72  std::for_each(json["wangtiles"].begin(), json["wangtiles"].end(), [&](const nlohmann::json &item) { m_wangTiles.emplace_back(item); });
73  if(json.count("cornercolors") > 0 && json["cornercolors"].is_array())
74  std::for_each(json["cornercolors"].begin(), json["cornercolors"].end(), [&](const nlohmann::json &item) { m_cornerColors.emplace_back(item); });
75  if(json.count("edgecolors") > 0 && json["edgecolors"].is_array())
76  std::for_each(json["edgecolors"].begin(), json["edgecolors"].end(), [&](const nlohmann::json &item) { m_edgeColors.emplace_back(item); });
77  if(json.count("properties") > 0 && json["properties"].is_array())
78  std::for_each(json["properties"].begin(), json["properties"].end(), [&](const nlohmann::json &item) { m_properties.add(item); });
79 
80  return allFound;
81 }
