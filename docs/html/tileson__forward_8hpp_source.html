59void tson::Tile::performDataCalculations()
60{
61 if(m_tileset == nullptr || m_map == nullptr)
62 return;
63
64 int firstId = m_tileset->getFirstgid(); //First tile id of the tileset
65 int columns = m_tileset->getColumns();
66 int rows = m_tileset->getTileCount() / columns;
67 int lastId = (m_tileset->getFirstgid() + m_tileset->getTileCount()) - 1;
68
69 int const gid = static_cast<int>(getGid());
70 if (gid >= firstId && gid <= lastId)
71 {
72 int const baseTilePosition = (gid - firstId);
73
74 int const tileModX = (baseTilePosition % columns);
75 int const currentRow = (baseTilePosition / columns);
76 int const offsetX = (tileModX != 0) ? ((tileModX) * m_map->getTileSize().x) : (0 * m_map->getTileSize().x);
77 int const offsetY = (currentRow < rows-1) ? (currentRow * m_map->getTileSize().y) : ((rows-1) * m_map->getTileSize().y);
78
79 tson::Vector2i spacing = m_tileset->getMarginSpacingOffset({tileModX, currentRow});
80 m_drawingRect = { offsetX + spacing.x, offsetY + spacing.y, m_map->getTileSize().x, m_map->getTileSize().y };
81 }
82 else
83 m_drawingRect = {0, 0, 0, 0};
84}

140void tson::Layer::decompressData()
141{
142
143 tson::DecompressorContainer *container = m_map->getDecompressors();
144 if(container->empty())
145 return;
146
147 if(m_encoding.empty() && m_compression.empty())
148 return;
149
150 std::string data = m_base64Data;
151 bool hasBeenDecoded = false;
152 if(!m_encoding.empty() && container->contains(m_encoding))
153 {
154 data = container->get(m_encoding)->decompress(data);
155 hasBeenDecoded = true;
156 }
157
158 if(!m_compression.empty() && container->contains(m_compression))
159 {
160 data = container->get(m_compression)->decompress(data);
161 }
162
163 if(hasBeenDecoded)
164 {
165 std::vector<uint8_t> bytes = tson::Tools::Base64DecodedStringToBytes(data);
166 m_data = tson::Tools::BytesToUnsignedInts(bytes);
167 }
168}

175bool tson::Layer::parse(IJson &json, tson::Map *map)
176{
177 m_map = map;
178
179 bool allFound = true;
180 if(json.count("tintcolor") > 0) m_tintColor = tson::Colori(json["tintcolor"].get<std::string>()); //Optional
181 if(json.count("compression") > 0) m_compression = json["compression"].get<std::string>(); //Optional
182 if(json.count("draworder") > 0) m_drawOrder = json["draworder"].get<std::string>(); //Optional
183 if(json.count("encoding") > 0) m_encoding = json["encoding"].get<std::string>(); //Optional
184 if(json.count("id") > 0) m_id = json["id"].get<int>(); //Optional
185 if(json.count("image") > 0) m_image = json["image"].get<std::string>(); //Optional
186 if(json.count("name") > 0) m_name = json["name"].get<std::string>(); else allFound = false;
187 if(json.count("offsetx") > 0 && json.count("offsety") > 0)
188 m_offset = {json["offsetx"].get<float>(), json["offsety"].get<float>()}; //Optional
189 if(json.count("opacity") > 0) m_opacity = json["opacity"].get<float>(); else allFound = false;
190 if(json.count("width") > 0 && json.count("height") > 0)
191 m_size = {json["width"].get<int>(), json["height"].get<int>()}; //else allFound = false; - Not mandatory for all layers!
192 if(json.count("transparentcolor") > 0) m_transparentColor = tson::Colori(json["transparentcolor"].get<std::string>()); //Optional
193 if(json.count("type") > 0) m_typeStr = json["type"].get<std::string>(); else allFound = false;
194 if(json.count("class") > 0) m_classType = json["class"].get<std::string>(); //Optional
195 if(json.count("visible") > 0) m_visible = json["visible"].get<bool>(); else allFound = false;
196 if(json.count("x") > 0) m_x = json["x"].get<int>(); else allFound = false;
197 if(json.count("y") > 0) m_y = json["y"].get<int>(); else allFound = false;
198 if(json.count("repeatx") > 0) m_repeatX = json["repeatx"].get<bool>(); //Optional
199 if(json.count("repeaty") > 0) m_repeatY = json["repeaty"].get<bool>(); //Optional
200
201 tson::Vector2f parallax {1.f, 1.f};
202 if(json.count("parallaxx") > 0)
203 parallax.x = json["parallaxx"].get<float>();
204 if(json.count("parallaxy") > 0)
205 parallax.y = json["parallaxy"].get<float>();
206
207 m_parallax = parallax;
208
209 //Handle DATA (Optional)
210 if(json.count("data") > 0)
211 {
212 if(json["data"].isArray())
213 {
214 auto &array = json.array("data");
215 std::for_each(array.begin(), array.end(), [&](std::unique_ptr<IJson> &item) { m_data.push_back(item->get<uint32_t>()); });
216 }
217 else
218 {
219 m_base64Data = json["data"].get<std::string>();
220 decompressData();
221 }
222 }
223
224 //More advanced data
225 if(json.count("chunks") > 0 && json["chunks"].isArray())
226 {
227 auto &chunks = json.array("chunks");
228 std::for_each(chunks.begin(), chunks.end(), [&](std::unique_ptr<IJson> &item) { m_chunks.emplace_back(*item); });
229 }
230 if(json.count("layers") > 0 && json["layers"].isArray())
231 {
232 auto &layers = json.array("layers");
233 std::for_each(layers.begin(), layers.end(), [&](std::unique_ptr<IJson> &item) { m_layers.emplace_back(*item, m_map); });
234 }
235 if(json.count("objects") > 0 && json["objects"].isArray())
236 {
237 auto &objects = json.array("objects");
238 std::for_each(objects.begin(), objects.end(), [&](std::unique_ptr<IJson> &item) { m_objects.emplace_back(*item, m_map); });
239 }
240 if(json.count("properties") > 0 && json["properties"].isArray())
241 {
242 auto &properties = json.array("properties");
243 tson::Project *project = (m_map != nullptr) ? m_map->getProject() : nullptr;
244 std::for_each(properties.begin(), properties.end(), [&](std::unique_ptr<IJson> &item) { m_properties.add(*item, project); });
245 }
246
247 setTypeByString();
248
249 return allFound;
250}

310void tson::Property::setValueByType(IJson &json)
311{
312 switch(m_type)
313 {
314 case Type::Color:
315 m_value = Colori(json.get<std::string>());
316 break;
317
318 case Type::File:
319 m_value = fs::path(json.get<std::string>());
320 break;
321
322 case Type::Int:
323 if(!m_propertyType.empty())
324 {
325 m_type = Type::Enum;
326 tson::EnumDefinition *def = (m_project != nullptr) ? m_project->getEnumDefinition(m_propertyType) : nullptr;
327 if(def != nullptr)
328 {
329 uint32_t v = json.get<uint32_t>();
330 m_value = tson::EnumValue(v, def);
331 }
332 else
333 m_value = tson::EnumValue();
334 }
335 else
336 m_value = json.get<int>();
337
338 break;
339
340 case Type::Boolean:
341 m_value = json.get<bool>();
342 break;
343
344 case Type::Float:
345 m_value = json.get<float>();
346 break;
347
348 case Type::String:
349 if(!m_propertyType.empty())
350 {
351 m_type = Type::Enum;
352 tson::EnumDefinition *def = (m_project != nullptr) ? m_project->getEnumDefinition(m_propertyType) : nullptr;
353 if(def != nullptr)
354 {
355 std::string v = json.get<std::string>();
356 m_value = tson::EnumValue(v, def);
357 }
358 else
359 m_value = tson::EnumValue();
360 }
361 else
362 setStrValue(json.get<std::string>());
363
364 break;
365
366 case Type::Class:
367 {
368 tson::TiledClass *baseClass = (m_project != nullptr) ? m_project->getClass(m_propertyType) : nullptr;
369 if (baseClass != nullptr)
370 {
371 tson::TiledClass c = *baseClass;
372 c.update(json);
373 m_value = c;
374 }
375 }
376 break;
377
378 case Type::Object:
379 m_value = json.get<uint32_t>();
380 break;
381 default:
382 setStrValue(json.get<std::string>());
383 break;
384 }
385}
